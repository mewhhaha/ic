module (!init: Init) where

declare effect Io {
  read: () => Text
  print: (bounded_borrow Text) => Unit
}

declare Init {
  io: Io
}

let Fx greet = () => {
  let (!Fx, name) = Fx.read()
  let (!Fx, ()) = Fx.print(borrow name)
  name
}

let result = greet()
return { result }

module () where

effect Counter {
  get: () => I32
  add: (I32) => Unit
}

let counter = {
  let count: I32 = 0
  Counter {
    get: (!resume) => !resume(count),
    add: (amount, !resume) => {
      count = count + amount
      !resume(())
    },
    return: value => { value, count },
  }
}

let make = () => Counter {
  get: (!resume) => !resume(0),
  return: value => value,
}

let result = try run() with counter
let (!left, !right) = dup !resume
return { result }
